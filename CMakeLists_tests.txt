cmake_minimum_required(VERSION 3.20)
project(HyperPredictTests LANGUAGES CXX)

set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(${CMAKE_SOURCE_DIR}/include)

    src/core/lockfree_queue.cpp
